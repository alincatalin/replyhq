Swift (iOS)
// AppDelegate.swift
import ReplyHQ

func application(_ application: UIApplication, didFinishLaunchingWithOptions...) {
  // Initialize ReplyHQ
  ReplyHQ.configure(
    apiKey: "rhq_live_xK7mP9nQ8wE2rT5vY4uH3sL6gF1dA0",
    userId: "user_12345"
  )
  
  // Enable shake-to-feedback (optional)
  ReplyHQ.enableShakeGesture = true
}
Pro Tip:
Set the user ID to match your internal user identifier. This lets you correlate ReplyHQ messages with your analytics.
